def predict_note(r):
    if 0>r or r>4.1:
        return None
    elif r>4:
        return "A+ (%.3g)"%(r)
    elif 4>=r>3.7:
        return "A (%.3g)"%(r)
    elif 3.7>=r>3.3:
        return "A- (%.3g)"%(r)
    elif 3.3>=r>3.0:
        return "B+ (%.3g)"%(r)    
    elif 3.0>=r>2.7:
        return "B (%.3g)"%(r)        
    elif 2.7>=r>2.3:
        return "B- (%.3g)"%(r)
    elif 2.3>=r>2.0:
        return "C+ (%.3g)"%(r)
    elif 2.0>=r>1.7:
        return "C (%.3g)"%(r)    
    elif 1.7>=r>1.3:
        return "C- (%.3g)"%(r)
    elif 1.3>=r>1.0:
        return "D+ (%.3g)"%(r)
    elif 1.0>=r>0.5:
        return "D (%.3g)"%(r)
    elif 0.5>=r>0.1:
        return "D- (%.3g)"%(r)
    elif 0.1>=r:
        return "F (%.3g)"%(r)
    else:
        raise ValueError
