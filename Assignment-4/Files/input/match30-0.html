class Loading(Frame):

    def __init__(self, root):

        Frame.__init__(self, root)


        button_opt = {'fill': Tkconstants.BOTH, 'padx': 5, 'pady': 5}



        Button(self, text='Load Past Student Grades', command=self.askopenfilenames).pack(**button_opt)
        Button(self, text='Load Your Current Transcript', command=self.askopenfilenames).pack(**button_opt)
        Button(self, text='See the Recommended Courses').pack(**button_opt)


        self.file_opt = options = {}
        options['defaultextension'] = '.txt'
        options['filetypes'] = [('all files', '.*'), ('text files', '.txt')]
        options['initialdir'] = 'C:\\'
        options['initialfile'] = 'myfile.txt'
        options['parent'] = root
        options['title'] = 'This is a title'

    def askopenfilenames(self):
        filename = tkFileDialog.askopenfilenames(**self.file_opt)

        if filename:
            return open(filename, 'r')


class App():
    def __init__(self, parent):
        self.parent = parent
        self.value_of_combo = 'Pearson'
        self.combo()

    def newselection(self, event):
        self.value_of_combo = self.box.get()


        print(self.value_of_combo)

    def combo(self):
        self.box_value = StringVar()
        self.box = ttk.Combobox(self.parent, textvariable=self.box_value)
        self.box.bind("<<ComboboxSelected>>", self.newselection)


        self.box['values'] = ('Pearson', 'Euclidean', 'Jaccard')
        self.box.current(0)
        self.box.grid(column=0, row=0)

    
if __name__ == '__main__':
    Loading(root).pack()
    root = Tk()
