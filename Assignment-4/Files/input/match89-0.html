dict={} #empty dictionary to store the data from excel documents

def askopenfilesGrade():
    #takes multiple excel files and convert to the dictionary
    #first store to the dict1 dictionary then store to dict dictionary
    data = tkFileDialog.askopenfilenames(filetypes=[("xlsx","*.xlsx")])
    for book in data:
        books = open_workbook(book)
        sheet = books.sheet_by_index(0)
        dict1={}
        for row in range(1, sheet.nrows):
            (code, title, grade) = (sheet.cell(row,col).value for col in range(sheet.ncols))
            dict1[(code,title)] = grade
            dict[book] = dict1
