def pastGradeClick(event):
    d = {}
    name = askopenfilename()
    book = xlrd.open_workbook(name)
    sheet = book.sheet_by_index(0)

    code = sheet.col_values(0)
    code.pop(0)

    title = sheet.col_values(1)
    title.pop(0)

    grade = sheet.col_values(2)
    grade.pop(0)

    for i in range(36):
        d[code[i],title[i]] = grade[i]
    print d
